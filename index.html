Install Arch

Show Disks and Partitions

lsblk

Enter Partition Tool in Disk sda

cgdisk /dev/sda

  • 500M ef00 boot
  • 4G 8200 swap
  • ALL 8300 arch

sda = disk1 ,sda1=disk1 part1 ,sda2 = disk1 part2

Format Partitions

mkfs.fat -F32 /dev/sda1

mkswap /dev/sda2

mkfs.ext4 /dev/sda3

Mount Partitions

mount /dev/sda3 /mnt

swapon /dev/sda2

mkdir -p /mnt/boot

mount /dev/sda1 /mnt/boot

Preinstall Packages

pacstrap /mnt base linux linux-firmware base-devel nano networkmanager

Create drives linkin points

genfstab -U /mnt >> /mnt/etc/fstab

Boot into arch instalation

arch-chroot /mnt

Inside ARCH install

Set local time

ln -sf /usr/share/zoneinfo/Europe/Bucharest /etc/localtime

hwclock --systohc

Set local date format , currency ,etc

nano /etc/locale.gen

  • ro_RO.UTF-8 UTF-8

locale-gen

Set language

nano /etc/locale.conf

  • LANG=en_US.UTF-8

locale-gen

Create root user

nano /etc/hostname

  • hostname
  • (hostname is irelevant , you login as root using root name)

passwd

("root" user password)

IDK Why

mkinitcpio -P

Install GRUB

pacman -Sy grub efibootmgr

grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=GRUB

grub-mkconfig -o /boot/grub/grub.cfg

Finished Basic ARCH Install

exit

unmount -R /mnt

reboot

Enable Wifi / Ethernet

systemctl enable NetworkManager

Add non-root User

useradd -g users -G wheel,storage,power -m [username]

passwd [username]

insted of [username] write the username you want

Enable Non-Root User To Use SUDO

nano ~/.bashrc

  • export EDITOR="nano"

sudo -E visudo

sudo nano /etc/sudoers

  • #Uncomment to allow members of group wheel to execute any command
  • %wheel ALL=(ALL) ALL

GIT/Github

Install GIT

sudo pacman -S git

GIT version

git --version

git version

Setup config

git config --global user.email "[user email]"

git config --global user.name "[name for commits ,not github username]"

Clone repo

git clone [github/AUR https link]

Check new changes

git status

Save changed file

git add [filename]

Commit changes locally

git commit -m "[commit comment]" [filename]

Save and commit locally all changes with [git comall] which means git commit all

git config --global alias.comall "git add -all && git commit -m

Push changes on remote

git push -u origin [branch name]
